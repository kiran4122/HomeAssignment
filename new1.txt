        } else {
            // Status not matched, keep checking until it matches or the wait count is reached
            int counter = 1;
            int endCounter = Integer.parseInt(TextProp.getProperty("waitCountForStatusChange"));
            while (!expectedNT6Status.contains(actualNT6Status.toLowerCase())) {
                base.response = RestAPI.getRestResponse(completeUrl, "GET");
                actualNT6Status = base.response.jsonPath().get("nt6Status");
                
                // Check if trade is rejected
                if (!expectedNT6Status.toLowerCase().contains("rejected")) {
                    if (actualNT6Status.toLowerCase().contains("rejected")) {
                        rejectedFlag = true;
                        overallRejectedFlag = true;
                        break;
                    }
                }

                // Increment counter and break loop if necessary
                if (counter >= endCounter) {
                    break;
                }
                counter++;
            }
        }
    } else {
        // Skip status check for FSIMCF3 and FSIMCF4
        Utility.logPass("Skipped status check for report type FSIMCF3 or FSIMCF4");
    }
